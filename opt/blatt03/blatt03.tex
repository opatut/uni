\item[\textbf{1.}]
    \begin{enumerate}
    \item[a)]
        \textbf{Starttableau}:

        \[\begin{array}{rcrcrcr}
            x_3 & = & 1 & + & 4 x_1 & - & x_2 \\
            x_4 & = & 2 & + &   x_1 & - & x_2 \\
            x_5 & = & 1 & - & \frac{1}{2} x_1 & + & x_2 \\ \hline
              z & = &   &   &   x_1 & + & 2 x_2 \\
        \end{array}\]

        Eingangsvariable: $x_2$, Ausgangsvariable: $x_3$

        \( x_2 = 1 + 4 x_1 - x_3 \) \\[3pt]
        \( x_4 = 2 + x_1 - (1 + 4 x_1 - x_3) = 1 - 3 x_1 + x_3 \) \\[3pt]
        \( x_5 = 1 - \frac{1}{2} x_1 + (1 + 4 x_1 - x_3) = 2 + \frac{7}{2} x_1 - x_3 \) \\[3pt]
        \( z = x_1 + 2 (1 + 4 x_1 - x_3) = 2 + 9x_1 - 2 x_3 \) \\[3pt]

        \textbf{Tableau nach 1. Iteration}:

        \[\begin{array}{rcrcrcr}
            x_2 & = & 1 & + & 4 x_1 & - & x_3 \\[3pt]
            x_4 & = & 1 & - & 3 x_1 & + & x_3 \\[3pt]
            x_5 & = & 2 & + & \frac{7}{2} x_1 & - & x_3 \\[3pt] \hline
              z & = & 2 & + & 9 x_1 & - & 2 x_3 \\
        \end{array}\]

        Eingangsvariable: $x_1$, Ausgangsvariable: $x_4$

        \( x_1 = \frac{1}{3} + \frac{1}{3} x_3 - \frac{1}{3} x_4 \) \\[3pt]
        \( x_2 = 1 + 4(\frac{1}{3} + \frac{1}{3} x_3 - \frac{1}{3} x_4) - x_3 = \frac{7}{3} + \frac{1}{3} x_3 - \frac{4}{3} x_4 \) \\[3pt]
        \( x_5 = 2 + \frac{7}{2} (\frac{1}{3} + \frac{1}{3} x_3 - \frac{1}{3} x_4) - x_3 = \frac{19}{6} + \frac{1}{6}x_3 - \frac{7}{6}x_4 \) \\[3pt]
        \( z = 2 + 9(\frac{1}{3} + \frac{1}{3} x_3 - \frac{1}{3} x_4) - 2 x_3 = 5 + x_3 - 3 x_4 \) \\[3pt]

        \textbf{Tableau nach 2. Iteration}:

        \[\begin{array}{rcrcrcr}
            x_1 & = & \frac{1}{3} & + & \frac{1}{3} x_3 & - & \frac{1}{3} x_4 \\[3pt]
            x_2 & = & \frac{7}{3} & + & \frac{1}{3} x_3 & - & \frac{4}{3} x_4 \\[3pt]
            x_5 & = & \frac{19}{6} & + & \frac{1}{6} x_3 & - & \frac{7}{3} x_4 \\[3pt] \hline
              z & = & 5 & + & x_3 & - & 3 x_4 \\
        \end{array}\]

        Eingangsvariable: $x_3$, Ausgangsvariable: keine möglich $\rightarrow$ unbeschränkt

        \( x_3 = t \Rightarrow \)
        \( x_1 = \frac{1}{3} + \frac{1}{3}t \), \( x_2 = \frac{7}{3} + \frac{1}{3}t \)

        \[ \vec{x} =
            \begin{pmatrix}\frac{1}{3}\\[3pt]\frac{7}{3}\end{pmatrix}
            + t \begin{pmatrix}\frac{1}{3}\\[3pt]\frac{1}{3}\end{pmatrix} \]

        \textbf{Zulässige Lösungen}:

        \[\begin{array}{lllllll}
            \text{Startlösung}:  & x_1 = 0 & x_2 = 0 & x_3 = 1 & x_4 = 2 & x_5 = 1 & z = 0 \\[3pt]
            \text{1. Iteration}: & x_1 = 0 & x_2 = 1 & x_3 = 0 & x_4 = 1 & x_5 = 2 & z = 2 \\[3pt]
            \text{2. Iteration}: & x_1 = \frac{1}{3} & x_2 = \frac{7}{3} & x_3 = 0 & x_4 = 0 & x_5 = \frac{19}{6} & z = 5 \\[3pt]
        \end{array}\]

    \item[b)]
        \textbf{Grafisches Verfahren}:

        \begin{tikzpicture}
            \path[fill = gray!30] (0,0)--(0, 1)--(1/3,7/3)--(3, 5)--(10, 5)--(10, 4)--(2,0)--cycle;

    \end{enumerate}

\item[\textbf{2.}]
    \begin{enumerate}
    \item[a)]
        \textbf{Hilfsproblem (als unzulässiges Tableau):}

        \[\begin{array}{rcrcrcrcr}
            x_3 & = & -4 & + & x_1 & + & x_2 & + & x_0 \\[3pt]
            x_4 & = & 2 & - & x_1 & - & 2 x_2 & + & x_0 \\[3pt]
            x_5 & = & -1 & + & x_1 & - & x_2 & + & x_0 \\[3pt] \hline
              w & = & & & & & & - & x_0\\
        \end{array}\]

        Eingangsvariable: $x_0$, Ausgangsvariable: $x_3$

        \( x_0 = 4 - x_1 - x_2 + x_3 \) \\[3pt]
        \( x_4 = 2 - x_1 - 2 x_2 + 4 - x_1 - x_2 + x_3 = 6 - 2 x_1 - 3x_2 + x_3 \) \\[3pt]
        \( x_5 = -1 + x_1 - x_2 + 4 - x_1 - x_2 + x_3 = 3 - 2x_2 + x_3 \) \\[3pt]
        \( w = -4 + x_1 + x_2 - x_3 \) \\[3pt]

        \textbf{Zulässiges Starttableau}:

        \[\begin{array}{rcrcrcrcr}
            x_0 & = & 4 & - & x_1 & - & x_2 & + & x_3 \\[3pt]
            x_4 & = & 6 & - & 2x_1 & - & 3x_2 & + & x_3 \\[3pt]
            x_5 & = & 3 & & & - & 2x_2 & + & x_3 \\[3pt] \hline
              w & = & -4& + & x_1 & + & x_2 & - & x_3\\
        \end{array}\]

        Eingangsvariable: $x_1$, Ausgangsvariable: $x_4$

        \( x_1 = 3 - \frac{3}{2} x_2 + \frac{1}{2} x_3 - \frac{1}{2} x_4 \) \\[3pt]
        \( x_0 = 4 - \left( 3 - \frac{3}{2}x_2 + \frac{1}{2}x_3 - \frac{1}{2}x_4 \right) - x_2 + x_3 =
            1 - \frac{5}{2} x_2 + \frac{3}{2}x_3 + \frac{1}{2}x_4 \) \\[3pt]
        \( x_5 = 3 - 2x_2 + x_3 \) \\[3pt]
        \( w = -4 + \left( 3 - \frac{3}{2}x_2 + \frac{1}{2}x_3 - \frac{1}{2}x_4 \right) + x_2 - x_3 =
            - 1 - \frac{1}{2} x_2 - \frac{1}{2}x_3 - \frac{1}{2}x_4 \) \\[3pt]

        \textbf{Tableau nach 1. Iteration}:

        \[\begin{array}{rcrcrcrcr}
            x_1 & = & 3 & - & \frac{3}{2} x_2 & + & \frac{1}{2} x_3 & - & \frac{1}{2} x_4 \\[3pt]
            x_0 & = & 1 & - & \frac{5}{2} x_2 & + & \frac{3}{2} x_3 & + & \frac{1}{2} x_4 \\[3pt]
            x_5 & = & 3 & & & - & 2x_2 & + & x_3 \\[3pt] \hline
              w & = & -1 & - & \frac{1}{2} x_2 & - & \frac{1}{2}x_3 & - & \frac{1}{2}x_4 \\[3pt]
        \end{array}\]

        Da $w, x_2, x_3, x_4 > 0$ gilt, ist dies unlösbar. Damit ist das
        Originalproblem unlösbar.

    \item[b)]
        \textbf{Hilfsproblem (als unzulässiges Tableau):}

        \[\begin{array}{rcrcrcrcr}
            x_3 & = & 8 & - & x_1 & + & x_2 & + & x_0 \\[3pt]
            x_4 & = & -3 & + & x_1 & + & x_2 & + & x_0 \\[3pt]
            x_5 & = & 1 & + & x_1 & - & 4x_2 & + & x_0 \\[3pt] \hline
              w & = & & & & & & - & x_0\\
        \end{array}\]

        Eingangsvariable: $x_0$, Ausgangsvariable: $x_4$

        \( x_0 = 3 - x_1 - x_2 - x_4 \) \\[3pt]
        \( x_3 = 8 - x_1 + x_2 + 3 - x_1 - x_2 - x_4 = 11 - 2 x_1 - x_4 \) \\[3pt]
        \( x_5 = 2 - 4x_2 + 3 - x_1 - x_2 - x_4 = 5  - 5 x_2 - x_4 \) \\[3pt]
        \( w = -3 + x_1 + x_2 + x_4 \) \\[3pt]

        \textbf{Zulässiges Starttableau}:

        \[\begin{array}{rcrcrcrcr}
            x_0 & = & 3 & - & x_1 & - & x_2 & - & x_4 \\[3pt]
            x_4 & = & 11 & - & 2x_1 & & & - & x_4 \\[3pt]
            x_5 & = & 5 & & & - & 5x_2 & - & x_4 \\[3pt] \hline
              w & = & -3& + & x_1 & + & x_2 & + & x_4\\
        \end{array}\]

        Eingangsvariable: $x_2$, Ausgangsvariable: $x_5$

        \( x_2 = 1 - \frac{1}{5} x_4 - \frac{1}{5} x_5 \) \\[3pt]
        \( x_0 = 3 - x_1 - \left(1 - \frac{1}{5}x_4 - \frac{1}{5}x_5 \right) - x_4 =
            2 - x_1 - \frac{4}{5} x_4 + \frac{1}{5}x_5 \) \\[3pt]
        \( x_3 = 11 - 2x_2 - x_4 \) \\[3pt]
        \( w = -3 + x_1 + 1 - \frac{1}{5}x_4 - \frac{1}{5}x_5 + x_4 =
            - 2 + x_1 + \frac{4}{5}x_4 - \frac{1}{5}x_5 \) \\[3pt]

        \textbf{Tableau nach 1. Iteration}:

        \[\begin{array}{rcrcrcrcr}
            x_2 & = & 1 & - & & & \frac{1}{5} x_4 & - & \frac{1}{5} x_5 \\[3pt]
            x_0 & = & 2 & - & x_1 & - & \frac{4}{5} x_4 & + & \frac{1}{5} x_5 \\[3pt]
            x_3 & = & 11 & & & - & 2x_2 & - & x_4 \\[3pt] \hline
              w & = & -2 & + & x_1 & + & \frac{4}{5} x_4 & - & \frac{1}{5}x_5 \\[3pt]
        \end{array}\]

        Eingangsvariable: $x_1$, Ausgangsvariable: $x_0$

        \( x_1 = 2 - \frac{4}{5}x_4 + \frac{1}{5}x_5 - x_0 \)\\[3pt]
        \( x_2 = 1 - \frac{1}{5}x_4 - \frac{1}{5}x_5 \)\\[3pt]
        \( x_3 = 11 - 2 \left( 2 - \frac{4}{5}x_4 + \frac{1}{5}x_5 - x_0 \right) - x_4 =
            7 + \frac{3}{5}x_4 - \frac{2}{5}x_5 + 2x_0 \)\\[3pt]
        \( w = -2 + 2 - \frac{4}{5}x_4 + \frac{1}{5}x_5 - x_0 + \frac{4}{5}x_4 - \frac{1}{5}x_5 = -x_0 \)\\[3pt]

        \textbf{Tableau nach 2. Iteration}:

        \[\begin{array}{rcrcrcrcr}
            x_1 & = & 2 & - & \frac{4}{5}x_4 & + & \frac{1}{5}x_5 & - & x_0 \\[3pt]
            x_2 & = & 1 & - & \frac{1}{5}x_4 & - & \frac{1}{5}x_5 & & \\[3pt]
            x_3 & = & 7 & + & \frac{3}{5}x_4 & - & \frac{2}{5}x_5 & + & 2x_0 \\[3pt] \hline
              w & = &  & & & & & & -x_0 \\[3pt]
        \end{array}\]

        Da $w, x_0 > 0$ gilt, ist dies unlösbar. Damit ist das Originalproblem
        unlösbar.

    \end{enumerate}
\end{enumerate}
